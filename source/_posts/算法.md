# 一、二分法

## 1 基础场景

场景1：在**非降序**数组中查找某个值的下标；

场景2：在**非降序**数组中查找某个值的起止位置。

## 2 常规代码

场景1：在非降序数组中查找某个值的下标

```c++
int search(vector<int>& nums, int target)
{
    int left = 0;
    int right = nums.size(); // 此时left和right的取值可以理解为是“左闭右开”区间：[left，right)
    while(left < right) // 1.写成<的原因：前面定义的取值范围是[left，right)。2. 此条件的终止条件是：left == right
    {
        int mid = left + (right - left) / 2;
        if(nums[mid] == target) // 找到了
        {
            return mid;
        }
        else if(nums[mid] < target) // 说明目标值在mid的右方，那么需要变更左边界
        {
            left = mid + 1; // 加1的原因：区间是左闭右开，走到此分支说明target的索引肯定大于mid
        }
        else if(nums[mid] > target) //说明目标值在mid的左方，那么需要变更右边界
        {
            right = mid; //不加1的原因：区间是左闭右开，right = mid，实际搜索的区间是[left， right-1]
        }
    }
    // 循环结束，终止条件为 left == right，即while外面left == right，走到这里说明没提前返回mid
    // 区间要么在最左边，要么在最右边
    if(left >= nums.size()) return -1; // 最右边，left的另一个含义：left左边有nums.size()个值不符合要求
    return nums[right] == target ? right : -1;  //最左边
}
```

```c++
class NumArray {

private: 
    vector<int> preArray;

public:
    NumArray(vector<int>& nums) 
    {
        preArray.resize(nums.size() + 1);
        for(int i = 1; i <= nums.size(); i++)
        {
            preArray[i] = preArray[i - 1] + nums[i - 1];
```

```c++
string minWindow(string s, string t) 
    {
        unordered_map<char, int> window, needs;
        for(char c : t) needs[c]++;
        int left=0, right=0, valid=0, start=0;
        int len = INT_MAX;
        while(right < s.size())
        {
            char a = s[right];
            right++;
            if(needs.count(a))
            {
                window[a]++;
                if(window[a] == needs[a])
                {
                    valid++;

            {
                if(right- left < len)
                {
                    start = left;
                    len = right - left;
                }
                char d = s[left];
                left++;
                if(needs.count(d))
                {
                    if(window[d] == needs[d])
                        valid--;
                    window[d]--;
                }
            }
        }
        return len == INT_MAX ? "" : s.substr(start, len);
    }
```

### **[64. 最小路径和](https://leetcode.cn/problems/minimum-path-sum/)**

**暴力解法**

```c++
// 思路：如果想知道从左下角到右下角路径的最小值，可以求出其子问题，也就是从[0,0] --> [i,j]路径的最小值，进一步，从[0,0] --> [i,j]路径的最小值就等于 从[0,0] --> [i-1,j-1]的最小值加上[i,j]的值。这样也就是递归的思路
//怎么想出来上面这个思路的？画个图，拆解一下问题，实例化进行分析
//暴力解法的入参填什么？不变量是grid，根据思路得到变量是横坐标和纵坐标即i和j，先确定这三个入参，进行【尝试】
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) 
    {
        // 暴力解法调用
        return minPathSum1(grid, grid.size() - 1, grid[0].size() - 1);
    }

    //暴力解法实现
    int minPathSum1(vector<vector<int>>& grid, int i, int j)
    {
        // base case: 只有一个格子，那么最小路径就是当前格子的值
        if(i == 0 && j == 0)
        {
            return grid[0][0];
        }
        int up = INT_MAX;
        int left = INT_MAX;
        // 行数大于1
        if(i > 0)
        {
            //得到从[i - 1, j]到[i, j]路径的最小值
            up = minPathSum1(grid, i - 1, j);
        }
        if(j > 0)
        {
            //得到从[i, j - 1]到[i, j]路径的最小值
            left = minPathSum1(grid, i, j - 1);
        }
        return min(up, left) + grid[i][j];
    }
};
```

**记忆化搜索**

```c++
// 记忆化搜索
// 就是加缓存表，那么缓存表的大小是多少？ 原格子中一个格子对应缓存表的一个格子，那么缓存表的大小就是原格子的大小
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) 
    {
		//创建一个缓存表，传到函数里
        vector<vector<int>> dp(grid.size(),vector<int>(grid[0].size(),-1));
        minPathSum2(grid, grid.size() - 1, grid[0].size() - 1, dp);
        return dp[grid.size() - 1][grid[0].size() - 1];
    }
	//记忆化搜索实现：根据暴力递归修改，基本上是固定写法
    int minPathSum2(vector<vector<int>>& grid, int i, int j, vector<vector<int>>& dp)
    {
        // 如果dp表的[i,j]位置不是-1，说明之前有计算过，那么直接返回缓存表里的值
		if(dp[i][j] != -1)
        {
            return dp[i][j]; // 用缓存
        }
        // dp表的[i,j]位置是-1，说明之前没加到缓存里，那么就需要加到缓存里
        int curVal = 0; //定义一个变量代表当前缓存表格子的值，算出来curval的之后更新到dp缓存表中
        if(i == 0 && j == 0)
        {
            curVal = grid[0][0];
        }
        else
        {
            int up = INT_MAX;
            int left = INT_MAX;
            // 行数大于1
            if(i > 0)
            {
                //得到从[i - 1, j]到[i, j]路径的最小值
                up = minPathSum2(grid, i - 1, j, dp);
            }
            if(j > 0)
            {
                //得到从[i, j - 1]到[i, j]路径的最小值
                left = minPathSum2(grid, i, j - 1, dp);
            }
            curVal = min(up, left) + grid[i][j];
        }
        dp[i][j] = curVal; //挂缓存
		return curVal; 
    }
};
```

**严格位置依赖的动态规划**

```c++
//其实就是创建一个dp表，dp表中每个格子的值dp[i][j]就是从dp[0][0] --> dp[i][j]的最小路径和，和缓存表存的内容一样，只是给dp表赋值的方式不一样
// 问题是怎么赋值？ 查看依赖关系，如果想得到dp[i][j]的值，那么就要知道dp[i-1][j]和dp[i][j-1]两者之间的最小值加上grid[i][j]的值，也就是左上角的值最好求，最终求出右下角的值
// 动态规划表的大小：每个可变参数的可能性数量相乘。例如本题中，有两个可变参数，行数和列数，可能性数量也就是取值，行数和列数的取值范围分别为 0--grid.size(),0--grid[0].size()那么动态规划表的大小就是 grid.size() * grid[0].size()
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) 
    {
		//创建一个dp表
        vector<vector<int>> dp(grid.size(),vector<int>(grid[0].size(),0));
        dp[0][0] = grid[0][0];
        // 第一行
        for(int index = 1; index < grid[0].size(); index++)
        {
            dp[0][index] = dp[0][index- 1] + grid[0][index];
        }
        // 第一列
        for(int index = 1; index < grid.size(); index++)
        {
            dp[index][0] = dp[index - 1][0] + grid[index][0];
        }
        // 算里面的
        for(int l = 1; l < grid.size(); l++)
        {
            for(int r = 1; r < grid[0].size(); r++)
            {
                dp[l][r] = min(dp[l - 1][r], dp[l][r - 1]) + grid[l][r];
            }
        }
        return dp[grid.size() - 1][grid[0].size() - 1];
    }
};
```

```c++
        //3 填dp表

                }
                else
                {
                    dp[i][j] = max(dp[i + 1][j], dp[i][j + 1]);
                }
            }
        }

            ans = max(func(t1, t2, i + 1, j), func(t1, t2, i, j + 1));

    int func2(vector<char>& t1, vector<char>& t2, int i, int j, vector<vector<int>>& dp)

    }
```

```c++
class Solution {
```

```c++
    int longestIncreasingPath(vector<vector<int>>& matrix) 
    {
        int n = matrix.size(), m = matrix[0].size();
        std::vector<std::vector<int>> dp(matrix.size(), std::vector<int>(matrix[0].size(), 0));
        int ans = 0;
        for(int i = 0; i < n; i++)
        {
            for(int j = 0; j < m; j++)
            {
                // ans = max(ans, func(matrix, i, j));
                ans = max(ans, func1(matrix, i, j, dp));
            }
        }
        return ans;
    }

    int func(vector<vector<int>>& matrix, int i, int j)
    {   

        int next = 0;
        if(i + 1 < matrix.size() && matrix[i + 1][j] > matrix[i][j])
        {
            next = max(next, func(matrix, i + 1, j));
        }
        if(i - 1 >= 0 && matrix[i - 1][j] > matrix[i][j])
        {
            next = max(next, func(matrix, i - 1, j));
        }
        if(j + 1 < matrix[0].size() && matrix[i][j + 1] > matrix[i][j])
        {
            next = max(next, func(matrix, i, j + 1));
        }
        if(j - 1 >= 0 && matrix[i][j - 1] > matrix[i][j])
        {
            next = max(next, func(matrix, i, j - 1));
        }
        return next + 1; // 4个next取最大再加上自己
    }
```

### [115. 不同的子序列](https://leetcode.cn/problems/distinct-subsequences/)

```c++
class Solution {
public:
    int numDistinct(string s, string t) 
    {
        vector<char> ss, tt;
        for(char it : s) ss.push_back(it);
        for(char it : t) tt.push_back(it);

        // 暴力解
        // return func(ss, tt, 0, 0); 

        // 加缓存
        // std::vector<std::vector<int>> dp(ss.size(), std::vector<int>(tt.size(), -1));
        // return func2(ss, tt, 0, 0, dp); 

        // 严格位置依赖的动态规划
        // 先确定格子大小
        int n = ss.size();
        int m = tt.size();
        std::vector<std::vector<uint64_t>> dp(ss.size() + 1, std::vector<uint64_t>(tt.size() + 1, 0));
        // 画图分析某个格子依赖哪些格子可以得到答案
        // 确定方向，从左到右/从右到左、从上到下/从下到上的方向（从简单到复杂的方向）比如本题中根据递归写法可以知道当前位置[i,j]格子的值依赖[i+1,j]和[i+1][j + 1]
        // 填dp表
        for(int i = n; i >= 0; i--)
        {
            dp[i][m] = 1;
        }
        for(int i = n - 1; i >= 0; i--)
        {
            for(int j = m - 1; j >= 0; j--)
            {
                if(ss[i] == tt[j])
                {
                    dp[i][j] = dp[i + 1][j + 1] + dp[i + 1][j];
                }
                else
                {
                    dp[i][j] = dp[i + 1][j];
                }
            }
        }
        return dp[0][0];
    }
    //总结：
    //base case可以尝试的方向：i,j索引分别为0和n
	// ss[i] == tt[j] 和 ss[i] != tt[j] 两者ans的求解方式与1143题.最长公共子序列 的形式一样，但ans求解方式不一样
    int func(vector<char>& ss, vector<char>& tt, int i , int j)
    {
        // if(i == ss.size() && j == tt.size()) return 1;
        // if(i != ss.size() && j == tt.size()) return 1;
        if(i == ss.size() && j != tt.size()) return 0;
        if(j == tt.size()) return 1;

        int ans = 0;
        if(ss[i] == tt[j])
        {
            // 加 func(ss, tt, i + 1, j) 的原因：字符匹配上了但是不用，用后面的字符来匹配
            ans = func(ss, tt, i + 1, j + 1) + func(ss, tt, i + 1, j); 
        }
        else
        {
            ans = func(ss, tt, i + 1, j);
        }
        return ans;
    }

    int func2(vector<char>& ss, vector<char>& tt, int i , int j, vector<vector<int>>& dp)
    {
        // if(i == ss.size() && j == tt.size()) return 1;
        // if(i != ss.size() && j == tt.size()) return 1;
        //base case可以尝试的方向：索引分别为0和n
        if(i == ss.size() && j != tt.size()) return 0;
        if(j == tt.size()) return 1;
        if(dp[i][j] != -1) return dp[i][j];
        int ans = 0;
        ans = (ss, tt, i + 1, j, dp);
        if(ss[i] == tt[j])
        {
            ans += func2(ss, tt, i + 1, j + 1, dp);
        }
        dp[i][j] = ans;
        return ans;
    }
};
```
